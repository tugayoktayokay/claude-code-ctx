'use strict';

const { DatabaseSync } = require('node:sqlite');
const fs   = require('fs');
const path = require('path');

const SCHEMA_V1 = `
CREATE TABLE IF NOT EXISTS snapshots (
  path        TEXT PRIMARY KEY,
  project     TEXT NOT NULL,
  name        TEXT NOT NULL,
  mtime       INTEGER NOT NULL,
  categories  TEXT,
  body        TEXT NOT NULL
);

CREATE INDEX IF NOT EXISTS idx_snapshots_project_mtime
  ON snapshots(project, mtime DESC);

CREATE VIRTUAL TABLE IF NOT EXISTS snapshots_fts USING fts5(
  body, categories,
  content='snapshots',
  content_rowid='rowid',
  tokenize='unicode61 remove_diacritics 2'
);

CREATE TRIGGER IF NOT EXISTS snapshots_ai AFTER INSERT ON snapshots BEGIN
  INSERT INTO snapshots_fts(rowid, body, categories)
    VALUES (new.rowid, new.body, new.categories);
END;

CREATE TRIGGER IF NOT EXISTS snapshots_ad AFTER DELETE ON snapshots BEGIN
  INSERT INTO snapshots_fts(snapshots_fts, rowid, body, categories)
    VALUES ('delete', old.rowid, old.body, old.categories);
END;

CREATE TRIGGER IF NOT EXISTS snapshots_au AFTER UPDATE ON snapshots BEGIN
  INSERT INTO snapshots_fts(snapshots_fts, rowid, body, categories)
    VALUES ('delete', old.rowid, old.body, old.categories);
  INSERT INTO snapshots_fts(rowid, body, categories)
    VALUES (new.rowid, new.body, new.categories);
END;

CREATE TABLE IF NOT EXISTS schema_version (v INTEGER PRIMARY KEY);
INSERT OR IGNORE INTO schema_version(v) VALUES (1);
`;

function openDb(dbPath) {
  fs.mkdirSync(path.dirname(dbPath), { recursive: true });
  let db;
  try {
    db = new DatabaseSync(dbPath);
    db.exec(SCHEMA_V1);
    return db;
  } catch (err) {
    if (db) { try { db.close(); } catch {} }
    if (String(err.message || '').includes('corrupt') || String(err.code || '').includes('CORRUPT')) {
      const stamp = new Date().toISOString().replace(/[:.]/g, '-');
      try { fs.renameSync(dbPath, `${dbPath}.corrupt-${stamp}`); } catch {}
      db = new DatabaseSync(dbPath);
      db.exec(SCHEMA_V1);
      return db;
    }
    throw err;
  }
}

function close(db) {
  try { db.close(); } catch {}
}

module.exports = { openDb, close };
```

function upsertSnapshot(db, { path: p, project, name, mtime, categories, body }) {
  const stmt = db.prepare(`
    INSERT INTO snapshots (path, project, name, mtime, categories, body)
    VALUES (?, ?, ?, ?, ?, ?)
    ON CONFLICT(path) DO UPDATE SET
      project    = excluded.project,
      name       = excluded.name,
      mtime      = excluded.mtime,
      categories = excluded.categories,
      body       = excluded.body
  `);
  stmt.run(p, project, name, mtime, categories || '', body);
}

function deleteMissing(db, project, keepPaths) {
  const rows = db.prepare('SELECT path FROM snapshots WHERE project = ?').all(project);
  const del = db.prepare('DELETE FROM snapshots WHERE path = ?');
  const tx = db.prepare('BEGIN IMMEDIATE');
  const commit = db.prepare('COMMIT');
  const rollback = db.prepare('ROLLBACK');
  tx.run();
  try {
    for (const { path: p } of rows) {
      if (!keepPaths.has(p)) del.run(p);
    }
    commit.run();
  } catch (err) {
    rollback.run();
    throw err;
  }
}
```

function buildFtsQuery(raw) {
  if (!raw) return '';
  const cleaned = String(raw)
    .replace(/["'()*:]/g, ' ')
    .toLowerCase()
    .split(/\s+/)
    .filter(t => t.length > 1);
  if (!cleaned.length) return '';
  return cleaned.map(t => `${t}*`).join(' ');
}

function ftsSearch(db, { project, query, limit = 50 }) {
  if (!query || !String(query).trim()) return [];
  const sql = `
    SELECT s.path, s.mtime, bm25(snapshots_fts) AS score
    FROM snapshots_fts
    JOIN snapshots s ON s.rowid = snapshots_fts.rowid
    WHERE snapshots_fts MATCH ? AND s.project = ?
    ORDER BY score
    LIMIT ?
  `;
  const tryRun = (q) => {
    if (!q) return null;
    try {
      return db.prepare(sql).all(q, project, limit);
    } catch (err) {
      if (/fts5|syntax error/i.test(String(err.message || ''))) return null;
      throw err;
    }
  };
  let rows = tryRun(String(query).trim());
  if (rows === null) rows = tryRun(buildFtsQuery(query));
  return rows || [];
}

'use strict';

const fs   = require('fs');
const path = require('path');
const { categorize } = require('./query.js');
const { USER_DIR }   = require('./config.js');
const fts            = require('./fts.js');

const DB_PATH = path.join(USER_DIR, 'index.db');

function readSnapshotHead(filePath) {
  try {
    const raw = fs.readFileSync(filePath, 'utf8');
    const endMatch = raw.match(/^---\n([\s\S]*?)\n---\n?([\s\S]*)$/);
    const fm = endMatch ? endMatch[1] : '';
    const body = endMatch ? endMatch[2] : raw;
    const meta = {};
    for (const line of fm.split('\n')) {
      const m = line.match(/^([a-zA-Z_][\w]*):\s*(.*)$/);
      if (!m) continue;
      meta[m[1]] = m[2].trim();
    }
    if (meta.categories) {
      meta.categories = meta.categories.replace(/^\[|\]$/g, '').split(',').map(s => s.trim()).filter(Boolean);
    } else {
      meta.categories = [];
    }
    return { meta, body, raw };
  } catch {
    return null;
  }
}

function projectKeyFromDir(memoryDir) {
  // memoryDir is ~/.claude/projects/<encoded>/memory — return <encoded>
  return path.basename(path.dirname(memoryDir)) || memoryDir;
}

function collectProjectCandidates(memoryDir, config) {
  if (!fs.existsSync(memoryDir)) return [];
  const maxCandidates = config?.retrieval?.max_candidates ?? 2000;
  let names;
  try { names = fs.readdirSync(memoryDir); } catch { return []; }
  const files = [];
  for (const name of names) {
    if (!name.startsWith('project_') || !name.endsWith('.md')) continue;
    const full = path.join(memoryDir, name);
    let stat;
    try { stat = fs.statSync(full); } catch { continue; }
    if (!stat.isFile()) continue;
    files.push({ name, path: full, mtime: stat.mtimeMs, size: stat.size });
  }
  files.sort((a, b) => b.mtime - a.mtime);
  const top = files.slice(0, maxCandidates);

  const project = projectKeyFromDir(memoryDir);
  const out = [];
  for (const f of top) {
    const head = readSnapshotHead(f.path);
    if (!head) continue;
    let categories = head.meta.categories;
    if (!categories.length) {
      const tokens = head.body.toLowerCase().split(/[^\p{L}\p{N}]+/u).filter(Boolean);
      categories = categorize(tokens, config?.categories || {});
    }
    out.push({
      name: f.name,
      path: f.path,
      project,
      mtime: f.mtime,
      size: f.size,
      categories,
      body: head.body,
      length: head.body.length || 1,
      meta: head.meta,
    });
  }
  return out;
}

function collectAllProjectsCandidates(projectsRoot, config) {
  if (!fs.existsSync(projectsRoot)) return [];
  let names;
  try { names = fs.readdirSync(projectsRoot); } catch { return []; }
  const all = [];
  for (const name of names) {
    const memDir = path.join(projectsRoot, name, 'memory');
    try {
      if (!fs.statSync(memDir).isDirectory()) continue;
    } catch { continue; }
    const part = collectProjectCandidates(memDir, config);
    for (const c of part) all.push(c);
  }
  all.sort((a, b) => b.mtime - a.mtime);
  const cap = config?.retrieval?.max_candidates ?? 2000;
  return all.slice(0, cap);
}

function syncFtsIndex(db, candidates) {
  if (!candidates.length) return;
  const byProject = new Map();
  for (const c of candidates) {
    if (!byProject.has(c.project)) byProject.set(c.project, []);
    byProject.get(c.project).push(c);
  }

  const existing = db.prepare('SELECT path, mtime FROM snapshots WHERE project = ?');
  const begin    = db.prepare('BEGIN IMMEDIATE');
  const commit   = db.prepare('COMMIT');
  const rollback = db.prepare('ROLLBACK');

  begin.run();
  try {
    for (const [project, list] of byProject) {
      const have = new Map();
      for (const r of existing.all(project)) have.set(r.path, r.mtime);
      for (const c of list) {
        const prev = have.get(c.path);
        if (prev === undefined || prev < Math.floor(c.mtime)) {
          fts.upsertSnapshot(db, {
            path: c.path,
            project,
            name: c.name,
            mtime: Math.floor(c.mtime),
            categories: (c.categories || []).join(','),
            body: c.body,
          });
        }
      }
      const keep = new Set(list.map(c => c.path));
      for (const p of have.keys()) {
        if (!keep.has(p)) {
          db.prepare('DELETE FROM snapshots WHERE path = ?').run(p);
        }
      }
    }
    commit.run();
  } catch (err) {
    rollback.run();
    throw err;
  }
}

function blendFtsScores(rows) {
  if (!rows.length) return new Map();
  const best  = rows[0].score;
  const worst = rows[rows.length - 1].score;
  const span  = Math.max(1e-9, worst - best);
  const out = new Map();
  for (const r of rows) {
    const normalized = rows.length === 1 ? 1 : 1 - (r.score - best) / span;
    out.set(r.path, normalized);
  }
  return out;
}

function scoreSnapshot(query, snap, config, ftsScores = null) {
  const weights   = config?.retrieval?.weights || { category: 0.4, bm25: 0.4, recency: 0.2 };
  const halfLife  = config?.retrieval?.recency_half_life_days || 90;

  const qCats = new Set(query.categories);
  const sCats = new Set(snap.categories || []);
  let inter = 0;
  for (const c of qCats) if (sCats.has(c)) inter++;
  const categoryScore = qCats.size === 0 ? 0 : inter / qCats.size;

  let keywordScore = 0;
  if (ftsScores && snap.path) {
    keywordScore = ftsScores.get(snap.path) || 0;
  } else if (query.nonStop.length) {
    const lower = (snap.body || '').toLowerCase();
    let total = 0;
    for (const term of query.nonStop) {
      const re = new RegExp(`\\b${term.replace(/[.*+?^${}()|[\]\\]/g, '\\$&')}\\b`, 'g');
      const matches = lower.match(re);
      if (matches) total += matches.length;
    }
    keywordScore = Math.min(total / Math.sqrt(Math.max(snap.length, 1)), 1.0);
  }

  const days = Math.max(0, (Date.now() - (snap.mtime || Date.now())) / 86_400_000);
  const recencyScore = Math.pow(2, -days / halfLife);

  const keywordWeight = weights.bm25 ?? weights.keyword ?? 0.3;
  const total = (weights.category ?? 0.4) * categoryScore
              + keywordWeight * keywordScore
              + (weights.recency ?? 0.2) * recencyScore;

  return {
    total,
    breakdown: { category: categoryScore, keyword: keywordScore, recency: recencyScore },
  };
}

function rank(query, candidates, config, { dbPath = DB_PATH } = {}) {
  const minScore = config?.retrieval?.min_score ?? 0.15;
  const topN     = config?.retrieval?.top_n     ?? 3;
  if (!candidates.length) return [];

  let ftsScores = null;
  let db = null;
  try {
    db = fts.openDb(dbPath);
    syncFtsIndex(db, candidates);
    const rawQuery = (query.raw && String(query.raw).trim())
      || (query.nonStop || []).join(' ');
    if (rawQuery) {
      const projects = new Set(candidates.map(c => c.project));
      const allRows = [];
      for (const project of projects) {
        const rows = fts.ftsSearch(db, { project, query: rawQuery, limit: 200 });
        for (const r of rows) allRows.push(r);
      }
      allRows.sort((a, b) => a.score - b.score);
      ftsScores = blendFtsScores(allRows);
    }
  } catch (err) {
    ftsScores = null;
  } finally {
    if (db) fts.close(db);
  }

  const scored = [];
  for (const c of candidates) {
    const s = scoreSnapshot(query, c, config, ftsScores);
    if (s.total < minScore) continue;
    scored.push({ snapshot: c, score: s.total, breakdown: s.breakdown });
  }
  scored.sort((a, b) => b.score - a.score);
  return scored.slice(0, topN);
}
